//自己实现字符串拷贝函数
char* StrCpy(char* src)
{
	//1、计算src的长度 len
	int len = 0;
	int j = 0;
	while (*(src+j)) {
		len++;
		j++;
	}	
	//2、用malloc申请长度为 len+1的内存空间
	char* newChar = (char*)malloc(sizeof(char) * (len + 1));
	//3、将src字符串逐个字符复制到新申请的内存空间
	int i = 0;
	for (; i < len + 1; i++) {
		newChar[i] = src[i];
	}
	//4、新申请内存空间的最后一位放入'\0'
	newChar[i] = '\0';
	//5、返回新申请的内存空间指针
	return newChar;
}
